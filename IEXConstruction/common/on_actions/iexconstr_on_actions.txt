
on_game_started = {
	on_actions = {
		iex_artisan_level_check
	}
}

on_yearly_pulse = {
	on_actions = {
		iex_artisan_level_check
	}
}

iex_artisan_level_check = {
	effect = {
		every_state = {
			remove_modifier = artisan_workshop_max_level
			set_local_variable = {
				name = iex_artisan_max_level_var
				value = iexconstr_artisan_max_value
			}
			add_modifier = {
				name = artisan_workshop_max_level
				multiplier = local_var:iex_artisan_max_level_var
			}
		}
	}
}
